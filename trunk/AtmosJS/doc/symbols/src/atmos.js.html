 75 //////////////////////////////////////////////////
 76 // dump function for use in debugging           //
 77 // (recursive)                                  //
 78 //////////////////////////////////////////////////
 79 function dumpObject( object, maxLevel ) {
 80     if ( typeof(maxLevel) == 'undefined' ) maxLevel = 1;
 81     if ( maxLevel < 0 ) return object; // we've reached our max depth
 82     var output = "[";
 83     for ( var property in object ) {
 84         if ( !object.hasOwnProperty( property ) ) continue;
 85         var value = object[property];
 86         if ( typeof(value) === 'object' && value != null ) value = dumpObject( value, maxLevel - 1 );
 87         output += property + "=" + value + ", ";
 88     }
 89     if ( output.length > 1 ) output = output.substr( 0, output.length - 2 );
 90     output += "]";
 91     return output;
 92 }

582 /**
583  * Creates a shareable URL that anyone (globally) can access.
584  *
585  * @param {String} id the object ID or path for which to generate the URL
586  * @param {Date} expirationDate the expiration date of the URL (as
587  * @param {String} disposition the content-disposition that should be specified in the response header for the shareable
588  *        URL. NOTE: this feature is only supported on Atmos 2.0.1 + hotfix XXXX
589  * @return a URL that can be used to share the object's content
590  */
591 AtmosRest.prototype.getShareableUrl = function( id, expirationDate, disposition ) {
592     if ( !expirationDate.getTime ) throw "expirationDate must be a Date object";
593 
594     var method = "GET";
595     var path = this._getPath( id );
596     var expires = Math.floor( expirationDate.getTime() / 1000 ); // convert to seconds
597 
598     // establish hash string for signing
599     var hashString = method + "\n" + path.toLowerCase() + "\n" + this.atmosConfig.uid + "\n" + expires;
600     if ( disposition ) hashString += "\n" + disposition;
601     this.info( "hash string:\n" + hashString );
602 
603     // sign hash string
604     var signature = this._doSignature( hashString, this.atmosConfig.secret );
605 
606     // generate query string
607     var query = "uid=" + encodeURIComponent( this.atmosConfig.uid );
608     query += "&expires=" + expires;
609     query += "&signature=" + encodeURIComponent( signature );
610     if ( disposition ) query += "&disposition=" + encodeURIComponent( disposition );
611 
612     var protocol, host;
613     if ( typeof(window) != 'undefined' ) {
614         protocol = window.location.protocol;
615         host = window.location.host;
616     } else {
617         protocol = this.atmosConfig.protocol;
618         host = this.atmosConfig.host;
619     }
620 
621     // compose URL
622     var url = protocol + "//" + host + this._encodeURI( path ) + "?" + query;
623     this.info( "Shareable URL: " + url );

1093 /**
1094  * Abstracts an ajax call
1095  * @param {Object} options the options for the call
1096  * options are:
1097  * method: HTTP method to use
1098  * uri: uri to call
1099  * headers: map/object of header name/value pairs
1100  * data: (optional) data to send
1101  * range: (optional) range of data to set in the target object
1102  * mimeType: (optional) mime type to use for data sent
1103  * state: (optional) state parameter to pass on to the callback under result.state
1104  * callback: async callback (completion) method
1105  * success: (optional) success method - if present, should create a result object and call the callback method
1106  * progress: (optional) callback for updating progress of an upload (applies only to PUT and POST requests)
1107  */
1108 AtmosRest.prototype._ajax = function( options ) {
1109     options.uri = this._resolveDots( options.uri );
1110 
1111     if ( options.headers ) {
1112         this._prepBaseHeaders( options.headers, options.range );
1113         if ( /(POST|PUT)/.test( options.method ) ) this._prepUploadHeaders( options.headers, options.mimeType );
1114 
1115         this._signRequest( options.method, options.headers, options.uri );
1116     }
1117 
1118     if ( !/^https?:\/\//.test( options.uri ) ) {
1119 
1120         // encode *after* signing
1121         options.uri = this._encodeURI( options.uri );
1122 
1123         // If a cross-domain request...
1124         if ( this.atmosConfig.host && this.atmosConfig.protocol ) {
1125             options.uri = this.atmosConfig.protocol + "//" + this.atmosConfig.host + options.uri;
1126         }
1127     }
1128 
1129     // success/error handling
1130     var xhr = this._getXMLHttpRequest(), me = this;
1131     xhr.onreadystatechange = function() {
1132         if ( xhr.readyState == 4 ) {
1133             if ( xhr.status < 400 ) {
1134                 if ( options.success ) options.success( xhr );
1135                 else options.callback( me._createResult( xhr, true, options.state ) );
1136             } else { // error
1137                 if ( options.error ) options.error( xhr );
1138                 else options.callback( me._createResult( xhr, false, options.state ) );
1139             }
1140         }
1141     };
1142 
1143     // progress callback
1144     try {
1145         if ( options.progress ) {
1146             (xhr.upload || xhr).onprogress = function( event ) {
1147                 if ( event.lengthComputable ) {
1148                     var progressPercent = Math.floor( (event.position || event.loaded) / (event.totalSize || event.total) * 100 );
1149                     options.progress( progressPercent );
1150                 }
1151             };
1152         }
1153     } catch( e ) {
1154         // progress isn't supported
1155     }
1156 
1157     xhr.open( options.method, options.uri, true );
1158 
1159     this._setHeaders( xhr, options.headers );
1160 
1161     if ( options.data ) xhr.send( options.data );
1162     else xhr.send();
1163 };

1183 /**
1184  * Resolves dot references in a path (parent/child/../ => parent/)
1185  * @param path the path to resolve
1186  */
1187 AtmosRest.prototype._resolveDots = function( path ) {
1188     if ( !path ) return path;
1189     var segments = path.split( "/" );
1190     for ( var i = 0; i < segments.length; i++ ) {
1191         if ( segments[i] == ".." ) {
1192             segments.splice( i - 1, 2 ); // remove this segment and the last
1193             i -= 2; // two items were removed, so make sure the index rewinds by two
1194         } else if ( segments[i] == "." ) {
1195             segments.splice( i--, 1 ); // remove only this segment and rewind the index by one
1196         }
1197     }
1198     return segments.join( "/" );
1199 };

1220     this.debug( "encodeURI: in: " + uri );
1221     var queryIndex = uri.indexOf( "?" );
1222     var query = "";
1223     if ( queryIndex != -1 ) {

1231     }

1236     }

1587  * @param {String} method the HTTP method (GET, PUT, DELETE, POST, HEAD)

1589  *        !!IMPORTANT!! this method assumes Content-Type and Range are already set!
1590  * @param {String} uri the path to the request
1591  */
1592 AtmosRest.prototype._signRequest = function( method, headers, uri ) {
1593     this.debug( this.atmosConfig.uid );
1594     this.debug( this.atmosConfig.secret );
1595 
1596     var mimeType = headers["Content-Type"];
1597     var range = headers["Range"];
1598 
1599     var emcheaders;
1600     if ( headers == "" ) {
1601         emcheaders = new Hash();
1602     } else {
1603         emcheaders = headers;
1604     }

1606     emcheaders["x-emc-uid"] = this.atmosConfig.uid;
1607

1609     this.debug( "HashString:\n" + hash_string );
1610 
1611     var signature = this._doSignature( hash_string, this.atmosConfig.secret );

1619 /**
1620  * Generates the string to sign
1621  * @param {String} method the HTTP method
1622  * @param {String} content_type the MIME type of the request body
1623  * @param {AtmosRange} range the HTTP range of the request, if needed
1624  * @param {String} date the HTTP Date header
1625  * @param {String} path the request uri
1626  * @param {Object} headers the object containing the HTTP headers as properties
1627  * @returns {String} the string to sign
1628  */
1629 AtmosRest.prototype._buildHashString = function( method, content_type, range, date, path, headers ) {
1630 
1631     var emcheaders = new Object();
1632     var string = "";
1633     string = method + "\n";
1634 
1635     if ( content_type ) {
1636         string += content_type + "\n";
1637     } else {
1638         string += "\n";
1639     }
1640     if ( range ) {
1641         string += range.toString() + "\n";
1642     } else {
1643         string += "\n";
1644     }
1645     if ( date ) {
1646         string += date + '\n';
1647     } else {
1648         string += "\n";
1649     }
1650 
1651     string += path.toLowerCase().trim() + "\n";
1652 
1653     for ( var prop in headers ) {
1654         if ( !headers.hasOwnProperty( prop ) ) continue;
1655         this.debug( "headers: prop: " + prop + " value: " + headers[prop] );
1656 
1657 
1658         var key = this._normalizeWS( prop.toLowerCase().trim() );
1659         if ( key.indexOf( "x-emc" ) != 0 ) {
1660             this.debug( "Skipping " + key );
1661             continue;
1662         }
1663 
1664         var value = headers[prop];
1665         if ( value ) value = this._normalizeWS( value.trim() );
1666         emcheaders[key] = value;
1667     }
1668 
1669     var keys = Object.keys( emcheaders );
1670     this.debug( "keys " + keys );
1671 
1672     keys.sort().forEach( function ( k ) {
1673         string += k + ":" + emcheaders[k] + "\n";
1674     } );
1675 
1676     return string.trim();
1677 };

1690 /**
1691  * Signs a string using HMAC-SHA1
1692  * @param {String} string the string content to sign
1693  * @param {String} secret the secret key, base-64 encoded.
1694  * @returns {String} the signature, base-64 encoded.
1695  */
1696 AtmosRest.prototype._doSignature = function( string, secret ) {
1697     this.debug( "Secret: " + secret );
1698 
1699     if ( isNodejs ) {
1700         var key = new Buffer( secret, 'base64' );
1701         var hmac = crypto.createHmac( "sha1", key.toString( 'binary' ) );
1702         hmac.update( string );
1703         return hmac.digest( 'base64' );
1704     } else {
1705         var sig = Crypto.HMAC( Crypto.SHA1, string, Crypto.util.base64ToBytes( secret ), {asBytes:true} );
1706         return Crypto.util.bytesToBase64( sig );
1707     }
1708 };
