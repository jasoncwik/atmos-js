 75 //////////////////////////////////////////////////
 76 // dump function for use in debugging           //
 77 // (recursive)                                  //
 78 //////////////////////////////////////////////////
 79 function dumpObject( object, maxLevel ) {
 80     if ( typeof(maxLevel) == 'undefined' ) maxLevel = 1;
 81     if ( maxLevel < 0 ) return object; // we've reached our max depth
 82     var output = "[";
 83     for ( var property in object ) {
 84         if ( !object.hasOwnProperty( property ) ) continue;
 85         var value = object[property];
 86         if ( typeof(value) === 'object' && value != null ) value = dumpObject( value, maxLevel - 1 );
 87         output += property + "=" + value + ", ";
 88     }
 89     if ( output.length > 1 ) output = output.substr( 0, output.length - 2 );
 90     output += "]";
 91     return output;
 92 }

331 /**
332  * Returns the information about the Atmos REST web service (right now, just the version number)
333  * @param {Object} state the user-defined state object passed to the callback as result.state
334  * @param {function} callback the completion callback (both error and success). The version number is contained in
335  *        result.value
336  */
337 AtmosRest.prototype.getServiceInformation = function( state, callback ) {
338     var me = this;
339     this._ajax( new AjaxRequest( {
340         method: 'GET',
341         uri: this.context + '/service',
342         headers: {},
343         state: state,
344         processResult: function( result, xhr ) {
345             if ( !result.success ) return;
346             var version = "n/a";
347 
348             // looking for /Version/Atmos (<Version><Atmos>X.X.X</Atmos></Version>)
349             var doc = me._getXmlDoc( xhr );
350             var versionNodes = doc.getElementsByTagName( "Version" );
351             if ( versionNodes.length ) version = me._getText( me._getChildByTagName( versionNodes[0], "Atmos" ) );
352 
353             result.value = new AtmosServiceInfo( version, false, false, false, false, false );
354 
355             var features = xhr.getResponseHeader( "x-emc-features" );
356             if ( features ) {
357                 var tokens = features.split( ", " );
358                 for ( var i = 0; i < tokens.length; i++ ) {
359                     tokens[i] = tokens[i].replaceAll( "-", "" ); // remove hyphens
360                     result.value[tokens[i]] = true; // serviceInfo[feature] = true
361                 }
362             } else {
363                 result.value.object = true;
364                 result.value.namespace = true;
365                 result.value.utf8 = xhr.getResponseHeader( "x-emc-support-utf8" ) == "true";
366             }
367         },
368         complete: callback
369     } ) );
370 };

665 /**
666  * Creates a shareable URL that anyone (globally) can access.
667  *
668  * @param {String} id the object ID or path for which to generate the URL
669  * @param {Date} expirationDate the expiration date of the URL (as
670  * @param {String} disposition the content-disposition that should be specified in the response header for the shareable
671  *        URL. NOTE: this feature must be supported by your Atmos version (check AtmosServiceInfo.browsercompat).
672  * @return a URL that can be used to share the object's content
673  */
674 AtmosRest.prototype.getShareableUrl = function( id, expirationDate, disposition ) {
675     if ( !expirationDate.getTime ) throw "expirationDate must be a Date object";
676 
677     var method = "GET";
678     var path = this._getPath( id );
679     var expires = Math.floor( expirationDate.getTime() / 1000 ); // convert to seconds
680 
681     // establish hash string for signing
682     var hashString = method + "\n" + path.toLowerCase() + "\n" + this.atmosConfig.uid + "\n" + expires;
683     if ( disposition ) hashString += "\n" + disposition;
684     this.info( "hash string:\n" + hashString );
685 
686     // sign hash string
687     var signature = this._doSignature( hashString, this.atmosConfig.secret );
688 
689     // generate query string
690     var query = "uid=" + encodeURIComponent( this.atmosConfig.uid );
691     query += "&expires=" + expires;
692     query += "&signature=" + encodeURIComponent( signature );
693     if ( disposition ) query += "&disposition=" + encodeURIComponent( disposition );
694 
695     var protocol, host;
696     if ( typeof(window) != 'undefined' ) {
697         protocol = window.location.protocol;
698         host = window.location.host;
699     } else {
700         protocol = this.atmosConfig.protocol;
701         host = this.atmosConfig.host;
702     }
703 
704     // compose URL
705     var url = protocol + "//" + host + this._encodeURI( path ) + "?" + query;
706     this.info( "Shareable URL: " + url );
707 
708     return url;

1385 /**
1386  * Resolves dot references in a path (parent/child/../ => parent/)
1387  * @param path the path to resolve
1388  */
1389 AtmosRest.prototype._resolveDots = function( path ) {
1390     if ( !path ) return path;
1391     var segments = path.split( "/" );
1392     for ( var i = 0; i < segments.length; i++ ) {
1393         if ( segments[i] == ".." ) {
1394             segments.splice( i - 1, 2 ); // remove this segment and the last
1395             i -= 2; // two items were removed, so make sure the index rewinds by two
1396         } else if ( segments[i] == "." ) {
1397             segments.splice( i--, 1 ); // remove only this segment and rewind the index by one
1398         }
1399     }
1400     return segments.join( "/" );

1819     emcheaders["x-emc-signature"] = signature;
1820 
1821     return signature;
1822 };

1824 /**
1825  * Generates the string to sign
1826  * @param {String} method the HTTP method
1827  * @param {String} content_type the MIME type of the request body
1828  * @param {AtmosRange} range the HTTP range of the request, if needed
1829  * @param {String} date the HTTP Date header
1830  * @param {String} path the request uri
1831  * @param {Object} headers the object containing the HTTP headers as properties
1832  * @returns {String} the string to sign
1833  */
1834 AtmosRest.prototype._buildHashString = function( method, content_type, range, date, path, headers ) {
1835 
1836     var emcheaders = new Object();
1837     var string = "";
1838     string = method + "\n";
1839 
1840     if ( content_type ) {
1841         string += content_type + "\n";
1842     } else {
1843         string += "\n";
1844     }
1845     if ( range ) {
1846         string += range.toString() + "\n";
1847     } else {
1848         string += "\n";
1849     }
1850     if ( date ) {
1851         string += date + '\n';
1852     } else {
1853         string += "\n";
1854     }
1855 
1856     string += path.toLowerCase().trim() + "\n";
1857 
1858     for ( var prop in headers ) {
1859         if ( !headers.hasOwnProperty( prop ) ) continue;
1860         this.debug( "headers: prop: " + prop + " value: " + headers[prop] );
1861 
1862 
1863         var key = this._normalizeWS( prop.toLowerCase().trim() );
1864         if ( key.indexOf( "x-emc" ) != 0 ) {
1865             this.debug( "Skipping " + key );
1866             continue;
1867         }
1868 
1869         var value = headers[prop];
1870         if ( value ) value = this._normalizeWS( value.trim() );
1871         emcheaders[key] = value;
1872     }
1873 
1874     var keys = Object.keys( emcheaders );
1875     this.debug( "keys " + keys );
1876 
1877     keys.sort().forEach( function( k ) {
1878         string += k + ":" + emcheaders[k] + "\n";
1879     } );
1880 
1881     return string.trim();
1882 };

1885  * Normalizes the whitespace in an object (condenses multiple spaces into one space)
1886  * @param {String} str the string to process
1887  * @returns {String} the output string
1888  */
1889 AtmosRest.prototype._normalizeWS = function( str ) {
1890     if ( str == null ) return null;
1891     str = str.replace( /\n/, " " );
1892     return str.replace( /\s+/, " " );
1893 };

1895 /**
1896  * Signs a string using HMAC-SHA1
1897  * @param {String} string the string content to sign
1898  * @param {String} secret the secret key, base-64 encoded.
1899  * @returns {String} the signature, base-64 encoded.
1900  */
1901 AtmosRest.prototype._doSignature = function( string, secret ) {
1902     this.debug( "Secret: " + secret );
1903 
1904     if ( isNodejs ) {
1905         var key = new Buffer( secret, 'base64' );
1906         var hmac = crypto.createHmac( "sha1", key.toString( 'binary' ) );
1907         hmac.update( string );
1908         return hmac.digest( 'base64' );
1909     } else {
1910         var sig = Crypto.HMAC( Crypto.SHA1, string, Crypto.util.base64ToBytes( secret ), {asBytes: true} );
1911         return Crypto.util.bytesToBase64( sig );
1912     }
1913 };
